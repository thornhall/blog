Most developers immediately reach for PostgreSQL when starting a new project. It's the default choice. But for 99% of projects, SQLite is actually faster, cheaper, and easier to maintain.

The Myth of Concurrency

People think SQLite can't handle concurrent writes. With WAL Mode (Write-Ahead Logging), this is largely solved for most read-heavy workloads.

"SQLite is not a toy database. It is a production-grade engine used by Apple, Google, and Facebook."

 74type Post struct {
 75	Title    string
 76	Slug     string
 77	Date     string
 78	Category string
 79	Excerpt  string
 80	Body     template.HTML
 81	Views    int
 82	Likes    int
 83}

 85type PageData struct {
 86	Title string
 87	Posts []Post
 88}

 90func main() {
 91	// 1. Configure Goldmark (Markdown Parser)
 92	md := goldmark.New(
 93		goldmark.WithExtensions(
 94			extension.GFM,
 95			meta.Meta,
 96			highlighting.NewHighlighting(
 97				// USE YOUR NEW CUSTOM STYLE HERE
 98				highlighting.WithStyle("custom-vscode"),
 99				highlighting.WithFormatOptions(
100				// Optional: Add line numbers if you want that "IDE" look
101				// html.WithLineNumbers(true),
102				),
103			),
104		),
105		goldmark.WithParserOptions(
106			parser.WithAutoHeadingID(),
107		),
108		goldmark.WithRendererOptions(
109			html.WithUnsafe(),
110		),
111	)
112
113	// 2. Read all .md files from the content folder
114	files, err := os.ReadDir("./content")
115	if err != nil {
116		log.Fatal("Could not read content directory:", err)
117	}
118
119	var posts []Post
120
121	for _, file := range files {
122		if filepath.Ext(file.Name()) != ".md" {
123			continue
124		}
125
126		// Read the file content
127		source, err := os.ReadFile("./content/" + file.Name())
128		if err != nil {
129			log.Fatal(err)
130		}
131
132		// 3. Convert Markdown to HTML & Extract Meta
133		var buf bytes.Buffer
134		context := parser.NewContext()
135		if err := md.Convert(source, &buf, parser.WithContext(context)); err != nil {
136			log.Fatal(err)
137		}
138
139		// Extract Metadata (Frontmatter)
140		metaData := meta.Get(context)
141
142		// Helper to safely get string values from the map
143		getString := func(key string) string {
144			if v, ok := metaData[key]; ok {
145				return fmt.Sprintf("%v", v)
146			}
147			return ""
148		}
149
150		p := Post{
151			Title:    getString("title"),
152			Slug:     getString("slug"),
153			Date:     getString("date"),
154			Category: getString("category"),
155			Excerpt:  getString("excerpt"),
156			Body:     template.HTML(buf.String()), // The converted HTML
157		}
158
159		posts = append(posts, p)
160	}
161
162	// 4. Prepare Templates
163	data := PageData{
164		Title: "The Obsidian Log",
165		Posts: posts,
166	}
167
168	tmplIndex, err := template.ParseFiles("templates/layout.html", "templates/index.html")
169	if err != nil {
170		log.Fatal(err)
171	}
172
173	tmplPost, err := template.ParseFiles("templates/layout.html", "templates/post.html")
174	if err != nil {
175		log.Fatal(err)
176	}
177
178	// 5. Build Index Page
179	f, err := os.Create("public/index.html")
180	if err != nil {
181		log.Fatal(err)
182	}
183	if err := tmplIndex.Execute(f, data); err != nil {
184		log.Fatal(err)
185	}
186	f.Close()
187	log.Println("Generated: public/index.html")
188
189	// 6. Build Article Pages
190	for _, post := range posts {
191		filename := "public/" + post.Slug + ".html"
192		f, err := os.Create(filename)
193		if err != nil {
194			log.Fatal(err)
195		}
196
197		if err := tmplPost.Execute(f, post); err != nil {
198			log.Fatal(err)
199		}
200		f.Close()
201		log.Println("Generated:", filename)
202	}
203}
